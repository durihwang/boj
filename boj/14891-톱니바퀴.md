## 코드
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

class Main {

    static int n;
    static int m;
    static int x;
    static int y;
    static int k;
    static int[] dx = {0, 0, -1, 1};
    static int[] dy = {1, -1, 0, 0};

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = 4;
        char[][] a = new char[n][8];
        for (int i = 0; i < n; i++) {
            String s = br.readLine();
            for (int j = 0; j < s.length(); j++) {
                a[i] = s.toCharArray();
            }
        }

        k = Integer.parseInt(br.readLine());
        for (int i = 0; i < k; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            int no = Integer.parseInt(st.nextToken()) - 1;
            int dir = Integer.parseInt(st.nextToken());
            int[] d = new int[n];
            d[no] = dir;

            // 현재 톱니 왼쪽을 회전해야 하는지 확인
            for (int j = no - 1; j >= 0; j--) {

                // 현재 톱니 오른쪽과 극이 다르면 회전해야 한다.
                if (a[j][2] != a[j + 1][6]) {
                    d[j] = -d[j + 1];
                } else {
                    // 현재 톱니가 회전하지 않으면 왼쪽 톱니도 회전하지 않는다.
                    break;
                }
            }

            // 현재 톱니 오른쪽을 회전해야 하는지 확인
            for (int j = no + 1; j < n; j++) {

                // 현재 톱니 왼쪽과 극이 다르면 회전해야 한다.
                if (a[j][6] != a[j - 1][2]) {
                    d[j] = -d[j - 1];
                } else {

                    // 현재 톱니가 회전하지 않으면 오른쪽 톱니도 회전하지 않는다.
                    break;
                }
            }

            for (int j = 0; j < n; j++) {
                if (d[j] == 0) {
                    continue;
                }

                if (d[j] == 1) {
                    char temp = a[j][7];
                    for (int l = 7; l > 0; l--) {
                        a[j][l] = a[j][l - 1];
                    }
                    a[j][0] = temp;
                } else if (d[j] == -1) {
                    char temp = a[j][0];
                    for (int l = 0; l < 7; l++) {
                        a[j][l] = a[j][l + 1];
                    }
                    a[j][7] = temp;
                }
            }
        }

        int answer = 0;
        for (int i = 0; i < n; i++) {
            if (a[i][0] == '1') {
                if (i == 0) {
                    answer += 1;
                } else if (i == 1) {
                    answer += 2;
                } else if (i == 2) {
                    answer += 4;
                } else {
                    answer += 8;
                }
            }
        }

        System.out.println(answer);

    }


}
```
